 function functionname (parameters)  //function declaration
 {
    //block of code to be executed    //function definiton

//Function example without return

function add(a,b,c) //function declaration
{
   console.log(a+b+c); //function definition
}
add(10,20,30) // function call    //a=10 b=20 c=30

}
